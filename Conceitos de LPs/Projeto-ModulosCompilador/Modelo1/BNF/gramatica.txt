<programa> ::= <instrucao> <instrucao>*
<instrucao> ::= <atribuicao> | <print>
<atribuicao> ::= <identificador> "=" <numero> ";"
<print> ::= "print " <identificador> ";"
<numero> ::= <digito> <digito>*
<digito> ::= [0-9]
<identificador> ::= <letra> <letra>*
<letra> ::= [a-z] | [A-Z]
